4.1 수동으로 이터레이터 소비¶

문제¶

해결¶

import os
import fnmatch
import gzip
import bz2
import re

def gen_find(filepat, top):
    '''
    Find all filenames in a directory tree that match a shell wildcard pattern
    '''
    for path, dirlist, filelist in os.walk(top):
        for name in fnmatch.filter(filelist, filepat):
            yield os.path.join(path,name)

def gen_opener(filenames):
    '''
    Open a sequence of filenames one at a time producing a file object.
    The file is closed immediately when proceeding to the next iteration. 
    '''
    for filename in filenames:
        if filename.endswith('.gz'):
            f = gzip.open(filename, 'rt')
        elif filename.endswith('.bz2'):
            f = bz2.open(filename, 'rt')
        else:
            f = open(filename, 'rt')
        yield f
        f.close()

def gen_concatenate(iterators):
    '''
    Chain a sequence of iterators together into a single sequence.
    '''
    for it in iterators:
        yield from it

def gen_grep(pattern, lines):
    '''
    Look for a regex pattern in a sequence of lines
    '''
    pat = re.compile(pattern)
    for line in lines:
        if pat.search(line):
            yield line

class Node:
    def __init__(self, value):
        self._value = value
        self._children = []
        
    def __repr__(self):
        return 'Node({!r})'.format(self._value)

토론¶

  • 파이썬의 이터레이터 프로토콜은 __iter__()가 실제 순환을 수행하기 위한 __next__() 메소드를 구현하는 특별 이터레이터 객체를 반환하기를 요구
  • 만약 다른 컨테이너에 들어있는 내용물에 대한 순환이 해야 할 작업의 전부라면, 이터레이터의 동작 방식을 완전히 이해할 필요는 없다. 이때는 요청 받은 순환을 다음으로 전달하기만 하면 됨

4.3 제너레이터로 새로운 순환 패턴 생성¶

문제¶

  • 내장 함수(range(), reversed())와는 다른 동작을 하는 순환 패턴을 만들고 싶다.

해결¶

  • 새로운 순환 패턴을 만들고 싶다면, generator 함수를 사용해서 정의해야 된다. 특정 범위의 부동 소수점 숫자를 만드는 generator 코드는 다음과 같다.
  • generator의 동작을 제대로 확인하려면 debugger 물려서 하나씩 따라가보자.
  • 이렇게 몇 번을 했는데도 아직 익숙해지지가 않네..

def frange(start, stop, step):
    x = start
    while x < stop:
        yield x
        x += step

4.4 이터레이터 프로토콜 구현¶

문제¶

  • 순환을 지원하는 객체를 만드는데, 이터레이터 프로토콜을 구현하는 쉬운 방법이 필요

해결¶

  • 객체에 대한 순환을 가장 쉽게 구현하는 방법은 제너레이터 함수를 사용하는 것
  • 트리 구조를 표현하기 위해 Node 클래스를 사용
  • 노드를 depth-first(깊이 우선) 패턴으로 순환하는 이터레이터를 구현하고 싶다면
In [1]:
class Node:
    def __init__(self, value):
        self._value = value
        self._children = []
        
    def __repr__(self):
        return 'Node({!r})'.format(self._value)
    
    def add_child(self, node):
        self._children.append(node)
        
    def __iter__(self):
        return iter(self._children)
    
    def depth_first(self):
        yield self
        for c in self:
            yield from c.depth_first()
            
if __name__ == '__main__':
    root = Node(0)
    child1 = Node(1)
    child2 = Node(2)
    root.add_child(child1)
    root.add_child(child2)
    child1.add_child(Node(3))
    child1.add_child(Node(4))
    child1.add_child(Node(5))
    
    for ch in root.depth_first():
        print(ch)

  • 이 코드에서 depth_first() 메소드는 직관적으로 읽고 이해할 수 있다. 처음에는 자기 자신을 만들고(yield) 그 다음에는 자식을 순환
  • 이때 그 자식은 depth_first() 메소드(yield from을 사용)로 아이템 만듦

Logic¶

  • Node(0)으로 root instance 생성
    • root는 _value가 0, _children = []
  • root.add_child(child1)을 하게 되면 _children list에 쌓이게 됨
  • root.depth_first()에서 yield self를 하게 되면 root의 self가 return이 되면서 흐름이 다시 for문으로 돌아옴
  • print(self)니까 __repr__로 이동하여 !r로(그냥 그대로의 모습) _value값 출력(0)
  • 그래서 최종적으로 Node(0)이 출력됨
  • 그 다음에 다시 for문 안에서 self를 돔
    • self가 돌 수 있는 이유는 __iter__ 때문인듯.. 없애고 다시 테스트 해보자.
    • 어쨋든 __iter__를 실행하면 _.children을 돌려준다.
    • 그렇게 되면 child1을 돌려받을 수 있고
    • yield from depth_first()를 다시 실행하게 된다.

class Node:
    def __init__(self, value):
        self._value = value
        self._children = []
        
    def __repr__(self):
        return 'Node({!r})'.format(self._value)
    
    def add_child(self, other_node):
        self._children.append(other_node)
        
    def __iter__(self):
        return iter(self._children)
    
    def depth_first(self):
        return DepthFirstIterator(self)
    
class DepthFirstIterator(object):
    '''
    Depth-first traversal
    '''
    def __init__(self, start_node):
        self._node = start_node
        self._children_iter = None
        self._child_iter = None
        
    def __iter__(self):
        return self
    
    def __next__(self):
        # 막 시작했다면 자신을 반환한다. 자식에 대해서 이터레이터를 생성한다.
        if self._children_iter is None:
            self._children_iter = iter(self._node)
            return self._node
        
        # 자식을 처리 중이라면 다음 아이템을 반환한다.
        elif self._child_iter:
            try:
                nextchild = next(self._child_iter)
                return nextchild
            except StopIteration:
                self._child_iter = None
                return next(self)
            
        # 다음 자식으로 진행하고 순환을 시작한다.
        else:
            self._child_iter = next(self._children_iter).depth_first()
            return next(self)

        n = 1
        while n <= self.start:

from collections import deque

class linehistory:
    def __init__(self, lines, histlen=3):
        self.lines = lines
        self.history = deque(maxlen=histlen)
        
    def __iter__(self):
        for lineno, line in enumerate(self.lines, 1):
            self.history.append((lineno, line))
            yield line
            
    def clear(self):
        self.history.clear()

import os
import fnmatch
import gzip
import bz2
import re

def gen_find(filepat, top):
    '''
    디렉토리 트리에서 와일드카드 패턴에 맻이하는 모든 파일 이름을 찾는다.
    '''
    for path, dirlist, filelist in os.walk(top):
        for name in fnmatch.filter(filelist, filepat):
            yield os.path.join(path, name)
            
def gen_opener(filenames):
    '''
    파일 이름 시퀀스를 하나씩 열어 파일 객체를 생성한다.
    다음 순환으로 넘어가는 순간 파일을 닫는다.
    '''
    for filename in filenames:
        if filename.endswith('.gz'):
            f = gzip.open(filename, 'rt')
        elif filename.endswith('.bz2'):
            f = bz2.open(filename, 'rt')
        else:
            f = open(filename, 'rt')
        yield f
        f.close()
        
def gen_concatenate(iterators):
    '''
    이터레이터 시퀀스를 합쳐 하나의 시퀀스로 만든다.
    '''
    for it in iterators:
        yield from it
        
def gen_grep(pattern, lines):
    '''
    라인 시퀀스에서 정규식 패턴을 살펴본다.
    '''
    pat = re.compile(pattern)
    for line in lines:
        if pat.search(line):
            yield line

yield 참고 사항¶

  • PythonEngine: 파이썬 yield (2) - yield는 언제 사용하는가.
  • yield from - What's New In Python 3.3 — Python 3.4.2rc1 documentation
  • PEP 380 yield from expression을 추가했다. generator가 다른 generator에게 연산 부분을 위임할 수 있도록 허락했다. 이것은 code의 section이 yield를 포함하게 허락했다.
  • 추가적으로 subgenerator는 value를 함께 돌려 받게된다. value는 위임하는 generator에 의해서 만들어진다.
  • subgenerator로 위임하는 첫번째 디자인원칙: yield from expression은 자의적인 subiterator에게 위임하는게 허락됐다.
  • 그러나, 일반적인 loop는 좋아하지 않는다. yield from은 subgenerator가 sent를 받기 위해 허락했다. 그리고 calling scope에서 직접 values를 보낸다. 그리고 final value는 outer generator에게 돌려준다.
  • 가장 중요한 원칙은 generator조차 허락하기 위해 존재한다. send, throw와 함께 사용되게끔 디자인됐다. multiplue subgenerator로 쉽게 single large function이 multiple subfunction 으로 나눠지는것처럼 흩어진다.

  • 내장 함수 iter()의 기능은 거의 알려져 있지 않음
  • 이 함수에는 선택적으로 인자없는 호출 가능 객체와 종료값을 입력으로 받음
  • 이렇게 사용하면 주어진 종료값을 반환하기 전까지 무한히 반복해서 호출 가능 객체를 호출
  • 이런 방식을 사용하면 입출력과 관련있는 반복 호출에 잘 동작
  • 예를 들어 소켓이나 파일에서 특정 크기의 데이터를 읽으려 한다면, 반복적으로 read()나 recv()를 호출하고 end-of-file을 확인해야 함

  • lambda를 사용하면 인자를 받지 않는 호출 객체를 만들 수 있고 원하는 크기의 인자를 recv()나 read()에 전달한다.
